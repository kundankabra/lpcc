MACRO   
INCR    &A,&B,&REG
MOVER   &REG,&A
ADDS1   &A,&B
MOVEM   &REG,&A
MEND
MACRO   
ADDS1   &C,&D
MOVER   AREG,&C
ADD AREG,&D
MOVEM   AREG,&D
WRITE   &D
MEND
MACRO   
SUBS1   &C,&D  
MOVER   BREG,&C
SUBS    BREG,&D
MOVEM   BREG,&D
WRITE   &D
MEND
START   200
READ    N1
READ    N2
ADDS1   N1,N2
SUBS1   N1,N2
READ    N3
READ    N4
INCR    N3,N4,DREG
STOP
N1  DS  2
N2  DS  2
N3  DS  2
N4  DS  2
END
